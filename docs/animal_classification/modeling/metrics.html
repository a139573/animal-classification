ðŸ“Š Model Visualization & Metrics

This module contains helper functions to generate diagnostic plots for model evaluation. It supports generating:

  • Confusion Matrices: To see where the model is confusing classes.
  • ROC Curves: To evaluate the trade-off between sensitivity and specificity.
  • Calibration Curves: To check if the model's confidence scores are reliable.

These functions handle the logic of switching between saving to disk (CLI mode) and returning PIL images (Dashboard mode).

 26def get_dataset_stats(data_dir: Path):
 27    """
 28    Calculates basic statistics for the dataset exploration tab.
 29
 30    Inspects the provided directory to count classes and filter valid image files
 31    to provide a summary of the dataset's composition.
 32
 33    Parameters
 34    ----------
 35    data_dir : Path
 36        The root directory containing animal class subfolders.
 37
 38    Returns
 39    -------
 40    tuple
 41        A tuple containing (pd.DataFrame of counts per species, 
 42        int total_number_of_classes, int total_image_count, 
 43        list_of_species_names).
 44    """
 45    class_folders = sorted([d for d in data_dir.iterdir() if d.is_dir()])
 46    counts = {d.name: len([f for f in d.glob('*') if f.suffix.lower() in {'.jpg', '.jpeg', '.png', '.bmp', '.webp'}]) for d in class_folders}
 47    df = pd.DataFrame(list(counts.items()), columns=['Species', 'Image Count'])
 48    return df, len(class_folders), sum(counts.values()), [d.name for d in class_folders]

 50def get_species_samples(data_dir: Path, species_name: str, num_samples=8):
 51    """
 52    Returns a list of random PIL images for a specific animal species. 
 53
 54    This function filters for common image extensions to prevent loading 
 55    metadata or hidden system files and provides samples for UI visualization.
 56
 57    Parameters
 58    ----------
 59    data_dir : Path
 60        The root directory of the dataset.
 61    species_name : str
 62        The folder name of the target species.
 63    num_samples : int, optional
 64        Maximum number of samples to retrieve (default: 8).
 65
 66    Returns
 67    -------
 68    list
 69        A list of PIL.Image objects corresponding to the sampled images.
 70    """
 71    species_path = data_dir / species_name
 72    if not species_path.exists():
 73        return []
 74    
 75    # Filter for valid image formats only
 76    valid_extensions = {'.jpg', '.jpeg', '.png', '.bmp', '.webp'}
 77    img_paths = [p for p in species_path.glob('*') if p.suffix.lower() in valid_extensions]
 78    
 79    selected = random.sample(img_paths, min(num_samples, len(img_paths)))
 80    return [Image.open(p).convert("RGB") for p in selected]

 82def plot_color_analysis(data_dir: Path):
 83    """
 84    Performs RGB profile analysis on a subset of classes for visualization.
 85
 86    Computes the average R, G, and B channel intensities across a sample of 
 87    species to visualize color dominance in different animal groups. 
 88    Limited to 25 species for plot readability.
 89
 90    Parameters
 91    ----------
 92    data_dir : Path
 93        The directory containing the class folders.
 94
 95    Returns
 96    -------
 97    PIL.Image.Image
 98        The generated stacked bar plot as a PIL image object.
 99    """
100    class_folders = sorted([d for d in data_dir.iterdir() if d.is_dir()])
101    rgb_means = []
102    names = []
103    
104    # We take a representative sample of 25 species for UI legibility
105    for folder in class_folders[:25]: 
106        imgs = [f for f in folder.glob('*') if f.suffix.lower() in {'.jpg', '.jpeg', '.png'}]
107        imgs = imgs[:10] # Sample 10 images for average
108        
109        if not imgs: continue
110        
111        class_rgb = []
112        for p in imgs:
113            with Image.open(p) as img:
114                arr = np.array(img.convert("RGB"))
115                class_rgb.append(arr.mean(axis=(0,1)))
116        
117        rgb_means.append(np.mean(class_rgb, axis=0))
118        names.append(folder.name)
119    
120    df = pd.DataFrame(rgb_means, columns=['R', 'G', 'B'], index=names) / 255.0
121    
122    fig, ax = plt.subplots(figsize=(10, 6))
123    df.plot(kind='bar', stacked=True, color=['#e74c3c', '#2ecc71', '#3498db'], ax=ax, alpha=0.8)
124    ax.set_title("Average Color Profiles (Subset of 25 Species)")
125    ax.set_ylabel("Mean Intensity")
126    plt.xticks(rotation=45, ha='right', fontsize=9)
127    plt.tight_layout()
128    
129    return fig_to_image(fig)

134def fig_to_image(fig=None):
135    """
136    Converts a Matplotlib figure into a PIL Image.
137    
138    This is used primarily by the Gradio dashboard to display plots 
139    in memory without saving them to disk.
140
141    Parameters
142    ----------
143    fig : matplotlib.figure.Figure, optional
144        The figure to convert. If None, uses the current active figure.
145    
146    Returns
147    -------
148    PIL.Image.Image
149        The plot rendered as an image object.
150    """
151    buf = io.BytesIO()
152    if fig:
153        fig.savefig(buf, format="png", bbox_inches='tight')
154    else:
155        plt.savefig(buf, format="png", bbox_inches='tight')
156    buf.seek(0)
157    img = Image.open(buf).copy()
158    plt.close(fig if fig else "all")
159    return img

161def plot_training_curves(train_loss, val_loss, val_acc, architecture_name="Model", output_path=None, is_demo=False):
162    """
163    Generates dual-axis convergence plots for training/validation performance.
164
165    Visualizes how loss decreases and accuracy increases over training epochs,
166    providing immediate feedback on model learning and potential overfitting.
167
168    Parameters
169    ----------
170    train_loss : list
171        Historical values of training loss per epoch.
172    val_loss : list
173        Historical values of validation loss per epoch.
174    val_acc : list
175        Historical values of validation accuracy per epoch.
176    architecture_name : str, optional
177        Name of the model backbone used (default: "Model").
178    output_path : Path, optional
179        Filesystem path to save the generated image (ignored if is_demo is True).
180    is_demo : bool, optional
181        If True, returns the plot as a PIL Image object (default: False).
182
183    Returns
184    -------
185    PIL.Image.Image or None
186        Returns the image object if in demo mode, otherwise None.
187    """
188    plt.style.use('seaborn-v0_8-muted')
189    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 5))
190    epochs = range(len(train_loss))
191    
192    # Subplot 1: Loss Convergence
193    ax1.plot(epochs, train_loss, 'o-', color='#1f77b4', linewidth=2, label='Train Loss')
194    ax1.plot(epochs, val_loss, 's--', color='#ff7f0e', linewidth=2, label='Val Loss')
195    ax1.set_title(f'Loss Convergence ({architecture_name.upper()})', fontsize=12, fontweight='bold')
196    ax1.set_xlabel('Epoch')
197    ax1.set_ylabel('Loss Value')
198    ax1.grid(True, linestyle='--', alpha=0.6)
199    ax1.legend()
200    
201    # Subplot 2: Accuracy Development
202    ax2.plot(epochs, val_acc, 'D-', color='#2ca02c', linewidth=2, label='Val Accuracy')
203    ax2.set_title(f'Accuracy Development ({architecture_name.upper()})', fontsize=12, fontweight='bold')
204    ax2.set_xlabel('Epoch')
205    ax2.set_ylabel('Accuracy (%)')
206    ax2.set_ylim(0, 1.05)
207    ax2.grid(True, linestyle='--', alpha=0.6)
208    ax2.legend(loc='lower right')
209    
210    plt.tight_layout()
211    
212    if is_demo:
213        return fig_to_image(fig)
214    if output_path:
215        fig.savefig(output_path / f"{architecture_name.lower()}_convergence.png", dpi=300)
216    plt.close(fig)
217    return None

219def plot_confusion_matrix(y_true, y_pred, architecture_name="Model", output_path=None, is_demo=False):
220    """
221    Generates a Confusion Matrix to visualize misclassifications.
222
223    Parameters
224    ----------
225    y_true : array-like
226        Ground truth (correct) target values.
227    y_pred : array-like
228        Estimated targets as returned by a classifier.
229    architecture_name : str
230        Name of the model (for the plot title).
231    output_path : Path, optional
232        Directory to save the `.png` file if `is_demo=False`.
233    is_demo : bool
234        If True, returns a PIL Image instead of saving to disk.
235
236    Returns
237    -------
238    PIL.Image.Image or numpy.ndarray
239        Returns the PIL Image if `is_demo=True`.
240        Returns the raw confusion matrix array if `is_demo=False`.
241    """
242    cm = confusion_matrix(y_true, y_pred)
243    fig, ax = plt.subplots(figsize=(6, 6))
244    im = ax.imshow(cm, cmap="Blues")
245    ax.set_title(f"Confusion Matrix ({architecture_name.upper()})")
246    fig.colorbar(im)
247    ax.set_xlabel("Predicted Label")
248    ax.set_ylabel("True Label")
249    fig.tight_layout()
250    
251    if is_demo:
252        return fig_to_image(fig)
253    if output_path:
254        fig.savefig(output_path / f"{architecture_name.lower()}_confusion_matrix.png")
255    plt.close(fig)
256    return cm

258def plot_roc_curves(y_true, y_probs, num_classes, architecture_name="Model", output_path=None, is_demo=False):
259    """
260    Generates Macro-Average ROC Curves.
261    
262    Calculates the Receiver Operating Characteristic (ROC) curve for each class 
263    and computes the macro-average AUC score.
264
265    Parameters
266    ----------
267    y_true : array-like
268        Ground truth labels.
269    y_probs : array-like
270        Probability estimates of the positive class.
271    num_classes : int
272        Total number of classes.
273    architecture_name : str
274        Name of the model.
275    output_path : Path, optional
276        Path to save the plot.
277    is_demo : bool
278        If True, returns the plot as an image and the AUC score.
279
280    Returns
281    -------
282    tuple
283        A tuple containing (PIL.Image.Image, float_auc_score) if `is_demo=True`.
284        A tuple containing (None, float_auc_score) if `is_demo=False`.
285    """
286    try:
287        fig, ax = plt.subplots(figsize=(8, 6))
288        y_true_onehot = np.eye(num_classes)[y_true]
289        
290        # Calculate AUC score
291        unique_labels = np.unique(y_true)
292        auc_score = roc_auc_score(
293            y_true_onehot[:, unique_labels],
294            y_probs[:, unique_labels],
295            average="macro",
296            multi_class="ovr"
297        )
298
299        fpr_interp = np.linspace(0, 1, 100)
300        tpr_all = []
301        
302        for i in range(num_classes):
303            y_true_i = y_true_onehot[:, i]
304            y_prob_i = y_probs[:, i]
305            # Skip classes with no positive samples in this batch
306            if np.sum(y_true_i) == 0 or np.sum(y_true_i) == len(y_true_i):
307                continue
308            
309            fpr_i, tpr_i, _ = roc_curve(y_true_i, y_prob_i)
310            ax.plot(fpr_i, tpr_i, color='blue', alpha=0.05)  # Very faint individual lines
311            tpr_all.append(np.interp(fpr_interp, fpr_i, tpr_i))
312
313        if tpr_all:
314            tpr_mean = np.mean(tpr_all, axis=0)
315            ax.plot(fpr_interp, tpr_mean, color='red', linewidth=2, label=f'Macro-average ROC (AUC={auc_score:.3f})')
316        
317        ax.plot([0, 1], [0, 1], "--", color="gray")
318        ax.set_xlabel("False Positive Rate")
319        ax.set_ylabel("True Positive Rate")
320        ax.set_title(f"ROC Curve ({architecture_name.upper()})")
321        ax.legend()
322        fig.tight_layout()
323        
324        if is_demo:
325            return fig_to_image(fig), auc_score
326        if output_path:
327            fig.savefig(output_path / f"{architecture_name.lower()}_roc_curve.png")
328        plt.close(fig)
329        return None, auc_score
330    except Exception as e:
331        print(f"âš ï¸ ROC plot failed: {e}")
332        return None, 0.0

334def plot_calibration_curve(y_true, y_probs, num_classes, architecture_name="Model", output_path=None, is_demo=False):
335    """
336    Generates a Reliability Diagram (Calibration Curve).
337    
338    Compares the model's predicted probability against the actual frequency of the class.
339    A perfectly calibrated model will follow the diagonal line.
340
341    Parameters
342    ----------
343    y_true : array-like
344        True labels.
345    y_probs : array-like
346        Predicted probabilities.
347    num_classes : int
348        Number of classes.
349    architecture_name : str
350        Name of the model.
351    output_path : Path, optional
352        Save location on the filesystem.
353    is_demo : bool
354        If True, returns the generated plot as a PIL image object.
355        
356    Returns
357    -------
358    PIL.Image.Image or None
359        The reliability diagram image object if is_demo is True, otherwise None.
360    """
361    try:
362        fig, ax = plt.subplots(figsize=(6, 6))
363        y_true_onehot = np.eye(num_classes)[y_true]
364        fixed_bins = np.linspace(0, 1, 10)
365        prob_true_all = []
366
367        for i in range(num_classes):
368            if np.sum(y_true_onehot[:, i]) == 0:
369                continue
370            prob_true, prob_pred = calibration_curve(y_true_onehot[:, i], y_probs[:, i], n_bins=10)
371            prob_true_all.append(np.interp(fixed_bins, prob_pred, prob_true))
372
373        if prob_true_all:
374            prob_true_mean = np.mean(prob_true_all, axis=0)
375            ax.plot(fixed_bins, prob_true_mean, marker="o", color="blue", label="Avg Calibration")
376
377        ax.plot([0, 1], [0, 1], "--", color="gray", label="Perfectly Calibrated")
378        ax.set_xlabel("Mean Predicted Probability")
379        ax.set_ylabel("Fraction of Positives")
380        ax.set_title(f"Calibration Plot ({architecture_name.upper()})")
381        ax.legend()
382        fig.tight_layout()
383        
384        if is_demo:
385            return fig_to_image(fig)
386        if output_path:
387            fig.savefig(output_path / f"{architecture_name.lower()}_calibration.png")
388        plt.close(fig)
389    except Exception as e:
390        print(f"âš ï¸ Calibration plot failed: {e}")
391        return None

260    """
261    Generates Macro-Average ROC Curves.
262    
263    Calculates the Receiver Operating Characteristic (ROC) curve for each class 
264    and computes the macro-average AUC score.
265
266    Parameters
267    ----------
268    y_true : array-like
269        Ground truth labels.
270    y_probs : array-like
271        Probability estimates of the positive class.
272    num_classes : int
273        Total number of classes.
274    architecture_name : str
275        Name of the model.
276    output_path : Path, optional
277        Path to save the plot.
278    is_demo : bool
279        If True, returns the plot as an image and the AUC score.
280
281    Returns
282    -------
283    tuple
284        A tuple containing (PIL.Image.Image, float_auc_score) if `is_demo=True`.
285        A tuple containing (None, float_auc_score) if `is_demo=False`.
286    """
287    try:
288        fig, ax = plt.subplots(figsize=(8, 6))
289        y_true_onehot = np.eye(num_classes)[y_true]
290        
291        # Calculate AUC score
292        unique_labels = np.unique(y_true)
293        auc_score = roc_auc_score(
294            y_true_onehot[:, unique_labels],
295            y_probs[:, unique_labels],
296            average="macro",
297            multi_class="ovr"
298        )
299
300        fpr_interp = np.linspace(0, 1, 100)
301        tpr_all = []
302        
303        for i in range(num_classes):
304            y_true_i = y_true_onehot[:, i]
305            y_prob_i = y_probs[:, i]
306            # Skip classes with no positive samples in this batch
307            if np.sum(y_true_i) == 0 or np.sum(y_true_i) == len(y_true_i):
308                continue
309            
310            fpr_i, tpr_i, _ = roc_curve(y_true_i, y_prob_i)
311            ax.plot(fpr_i, tpr_i, color='blue', alpha=0.05)  # Very faint individual lines
312            tpr_all.append(np.interp(fpr_interp, fpr_i, tpr_i))
313
314        if tpr_all:
315            tpr_mean = np.mean(tpr_all, axis=0)
316            ax.plot(fpr_interp, tpr_mean, color='red', linewidth=2, label=f'Macro-average ROC (AUC={auc_score:.3f})')
317        
318        ax.plot([0, 1], [0, 1], "--", color="gray")
319        ax.set_xlabel("False Positive Rate")
320        ax.set_ylabel("True Positive Rate")
321        ax.set_title(f"ROC Curve ({architecture_name.upper()})")
322        ax.legend()
323        fig.tight_layout()
324        
325        if is_demo:
326            return fig_to_image(fig), auc_score
327        if output_path:
328            fig.savefig(output_path / f"{architecture_name.lower()}_roc_curve.png")
329        plt.close(fig)

331    except Exception as e:
332        print(f"âš ï¸ ROC plot failed: {e}")
333        return None, 0.0
